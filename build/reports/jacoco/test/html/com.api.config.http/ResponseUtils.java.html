package com.api.config.http;

import org.springframework.http.HttpStatus;
import org.springframework.http.MediaType;
import org.springframework.http.ResponseEntity;

/**
 * fileName       : ResEntity
 * author         : crlee
 * date           : 2023/08/21
 * description    :
 * ===========================================================
 * DATE              AUTHOR             NOTE
 * -----------------------------------------------------------
 * 2023/08/21        crlee       최초 생성
 */

public class ResponseUtils {

    public static ResponseEntity<Message> ResEntity(String msg,Object data,HttpStatus httpStatus){
        Message res = new Message(msg, data);
        return buildResponseEntity(res, httpStatus);
    }
    public static ResponseEntity<Message> ResEntity(String msg,Object data){
        Message res = new Message(msg, data);
        return buildResponseEntity(res, HttpStatus.OK);
    }
    public static ResponseEntity<Message> ResEntity(String msg){
        Message res = new Message(msg);
        return buildResponseEntity(res, HttpStatus.OK);
    }
    public static ResponseEntity<Message> ResEntity(String msg,HttpStatus httpStatus){
        Message res = new Message(msg);
        return buildResponseEntity(res, httpStatus);
    }
    private static ResponseEntity<Message> buildResponseEntity(Message message, HttpStatus httpStatus) {
        return ResponseEntity
                .status(httpStatus)
                .contentType(MediaType.APPLICATION_JSON)
                .body(message);
    }
}
